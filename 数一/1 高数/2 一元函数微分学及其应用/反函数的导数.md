$$
\varphi'(y) = \frac{dx}{dy} = \frac{1}{\dfrac{dy}{dx}} = \frac{1}{f'(x)},
$$
$$
\varphi''(y) = \frac{d}{dx}\left[ \frac{1}{f'(x)} \right] \cdot \frac{dx}{dy} = -\frac{f''(x)}{\left[ f'(x) \right]^2} \cdot \frac{1}{f'(x)}.
$$
