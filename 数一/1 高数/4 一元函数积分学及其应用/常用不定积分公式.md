> [!tip] 常用不定积分公式
> $$\begin{align}  & \int \dfrac{1}{x} \, dx = \ln|x| + C \\[6pt]  & \begin{aligned}   &\int \tan x \, dx = -\ln|\cos x| + C; \quad \int \cot x \, dx = \ln|\sin x| + C; \\  &\int \sec x \, dx = \ln|\sec x + \tan x| + C; \quad \int \csc x \, dx = \ln|\csc x - \cot x| + C; \\  &\int \sec^2 x \, dx = \tan x + C; \quad \int \csc^2 x \, dx = -\cot x + C; \\  &\int \sec x \tan x \, dx = \sec x + C; \quad \int \csc x \cot x \, dx = -\csc x + C \end{aligned} \\[6pt] & \begin{cases} \int \dfrac{1}{1+x^2} \, dx = \arctan x + C, \\[6pt] \int \dfrac{1}{a^2+x^2} \, dx = \dfrac{1}{a}\arctan \dfrac{x}{a} + C \quad (a > 0) \end{cases} \\[8pt] &\begin{cases} \int \dfrac{1}{\sqrt{1-x^2}} \, dx = \arcsin x + C, \\[6pt] \int \dfrac{1}{\sqrt{a^2-x^2}} \, dx = \arcsin \dfrac{x}{a} + C \quad (a > 0) \end{cases} \\[8pt] &\begin{cases} \int \dfrac{1}{\sqrt{x^2+a^2}} \, dx = \ln\left(x + \sqrt{x^2+a^2}\right) + C \quad (\text{常见 } a=1), \\[6pt] \int \dfrac{1}{\sqrt{x^2-a^2}} \, dx = \ln\left| x + \sqrt{x^2-a^2} \right| + C \quad (|x| > |a|) \end{cases} \\[8pt] &\int \dfrac{1}{x^2-a^2} \, dx = \dfrac{1}{2a}\ln\left| \dfrac{x-a}{x+a} \right| + C \quad \left( \int \dfrac{1}{a^2-x^2} \, dx = \dfrac{1}{2a}\ln\left| \dfrac{x+a}{x-a} \right| + C \right) \\[8pt] & \int \sqrt{a^2-x^2} \, dx = \dfrac{a^2}{2}\arcsin \dfrac{x}{a} + \dfrac{x}{2}\sqrt{a^2-x^2} + C \quad (a > |x| \geq 0) \\[6pt] &\begin{aligned} \int \sin^2 x \, dx &= \dfrac{x}{2} - \dfrac{\sin 2x}{4} + C \quad \left( \sin^2 x = \dfrac{1-\cos 2x}{2} \right); \\[4pt] \int \cos^2 x \, dx &= \dfrac{x}{2} + \dfrac{\sin 2x}{4} + C \quad \left( \cos^2 x = \dfrac{1+\cos 2x}{2} \right); \\[4pt] \int \tan^2 x \, dx &= \tan x - x + C \quad (\tan^2 x = \sec^2 x - 1); \\[4pt] \int \cot^2 x \, dx &= -\cot x - x + C \quad (\cot^2 x = \csc^2 x - 1) \end{aligned} \end{align}$$^d9zo0z
