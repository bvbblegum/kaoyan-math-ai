$$\begin{align} \tag{1} \frac{1}{1-x} &= 1 + x + x^2 + \cdots + x^n + \cdots = \sum_{n=0}^{\infty} x^n \quad (-1 < x < 1), \\ \tag{2} \frac{1}{1+x} &= 1 - x + x^2 - \cdots + (-1)^n x^n + \cdots = \sum_{n=0}^{\infty} (-1)^n x^n \quad (-1 < x < 1), \\ \tag{3} e^x &= 1 + x + \frac{x^2}{2!} + \cdots + \frac{x^n}{n!} + \cdots = \sum_{n=0}^{\infty} \frac{x^n}{n!} \quad (-\infty < x < +\infty), \\ \tag{4} \sin x &= x - \frac{x^3}{3!} + \cdots + \frac{(-1)^n x^{2n+1}}{(2n+1)!} + \cdots = \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n+1}}{(2n+1)!} \quad (-\infty < x < +\infty), \\ \tag{5} \cos x &= 1 - \frac{x^2}{2!} + \cdots + \frac{(-1)^n x^{2n}}{(2n)!} + \cdots = \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n}}{(2n)!} \quad (-\infty < x < +\infty), \\ \tag{6} \ln(1+x) &= x - \frac{x^2}{2} + \cdots + \frac{(-1)^{n-1} x^n}{n} + \cdots = \sum_{n=1}^{\infty} \frac{(-1)^{n-1} x^n}{n} \quad (-1 < x \leq 1), \\ \tag{7}  (1+x)^\alpha &= 1 + \alpha x + \frac{\alpha(\alpha-1)}{2!} x^2 + \cdots + \frac{\alpha(\alpha-1)\cdots(\alpha-n+1)}{n!} x^n + \cdots \\ &= \sum_{n=0}^{\infty} \frac{\alpha(\alpha-1)\cdots(\alpha-n+1)}{n!} x^n  \quad (-1 < x < 1,\ \text{区间端点展开式是否成立由 } \alpha \text{ 的值确立}) \end{align}
$$
